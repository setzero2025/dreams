## 2. 架构设计原则

### 2.1 核心原则

1. **前后端分离**：前端专注UI/UX交互，后端专注业务逻辑与AI能力编排
2. **跨平台一致性**：一套代码库支撑iOS/Android双平台，保持95%+代码复用率
3. **云本地协同**：支持离线使用，联网后自动同步，数据不丢失
4. **AI能力解耦**：各AI服务（语音/生图/生视频/剧本）独立封装，便于替换和扩展
5. **安全优先**：JWT认证、数据加密、敏感信息脱敏

### 2.2 技术选型考量

| 考量维度 | 决策依据 |
|---------|---------|
| **开发效率** | 选择生态成熟、社区活跃的技术栈 |
| **性能要求** | AI生成任务异步处理，前端保持响应 |
| **维护成本** | 统一技术栈，减少团队学习成本 |
| **扩展性** | 微服务化设计，支持后续功能迭代 |

### 4.2 核心模块设计

#### 4.2.1 离线优先架构

```typescript
// 离线同步策略
interface SyncStrategy {
  // 本地SQLite作为数据源，网络请求作为增强
  localFirst: boolean;
  
  // 操作队列管理
  pendingQueue: {
    type: 'CREATE' | 'UPDATE' | 'DELETE';
    entity: 'dream' | 'media' | 'user';
    data: unknown;
    timestamp: number;
    retryCount: number;
  }[];
  
  // 自动同步触发条件
  syncTriggers: {
    onNetworkRestore: boolean;    // 网络恢复时
    onAppForeground: boolean;     // 应用前台时
    onUserAction: boolean;        // 用户主动触发
    interval: number;             // 定时同步间隔(ms)
  };
}
```

#### 4.2.2 AI生成状态机

```typescript
// AI生成任务状态管理
type GenerationState = 
  | 'idle'           // 空闲
  | 'queued'         // 排队中
  | 'processing'     // 处理中
  | 'partial'        // 部分完成（长视频分阶段）
  | 'completed'      // 完成
  | 'failed'         // 失败
  | 'cancelled';     // 已取消

interface AIGenerationTask {
  id: string;
  type: 'image' | 'video' | 'story' | 'interpretation';
  state: GenerationState;
  progress: number;           // 0-100
  currentStep?: string;       // 当前步骤描述
  modelSource: string;        // 模型来源
  estimatedTime?: number;     // 预计剩余时间(秒)
  result?: unknown;
  error?: {
    code: string;
    message: string;
    retryable: boolean;
  };
}
```

#### 5.2.1 AI编排服务

AI编排服务负责协调多个AI服务商，管理生成任务的完整生命周期：

```typescript
// AI编排服务核心逻辑
class AIOrchestrationService {
  constructor(
    private taskQueue: TaskQueueService,
    private quotaService: QuotaService,
    private aiProviders: {
      xunfei: XunfeiASRService;
      doubao: DoubaoImageService;
      alibaba: AlibabaVideoService;
      kimi: KimiScriptService;
    }
  ) {}

  // 梦生图片 - 调用豆包
  async generateImage(params: ImageGenerationParams): Promise<Task> {
    // 1. 检查用户额度
    await this.quotaService.checkQuota(params.userId, 'image');
    
    // 2. 创建任务记录
    const task = await this.taskQueue.createTask({
      type: 'image',
      userId: params.userId,
      dreamId: params.dreamId,
      status: 'queued',
      params,
    });
    
    // 3. 加入处理队列
    await this.taskQueue.add('image-generation', {
      taskId: task.id,
      prompt: this.buildImagePrompt(params),
      style: params.style,
    });
    
    return task;
  }

  // 梦生视频 - 调用阿里百炼
  async generateVideo(params: VideoGenerationParams): Promise<Task> {
    await this.quotaService.checkQuota(params.userId, 'video');
    
    const task = await this.taskQueue.createTask({
      type: 'video',
      userId: params.userId,
      dreamId: params.dreamId,
      status: 'queued',
      params,
    });
    
    await this.taskQueue.add('video-generation', {
      taskId: task.id,
      prompt: this.buildVideoPrompt(params),
      ratio: params.ratio,
      duration: 5, // 默认5秒
    });
    
    return task;
  }

  // 梦境剧情 - 多阶段编排
  async generateStory(params: StoryGenerationParams): Promise<Task> {
    await this.quotaService.checkQuota(params.userId, 'story');
    
    const task = await this.taskQueue.createTask({
      type: 'story',
      userId: params.userId,
      dreamId: params.dreamId,
      status: 'queued',
      params,
    });
    
    // 长视频生成分为多个阶段：
    // 1. 生成剧本 (Kimi)
    // 2. 生成关键帧图 (豆包)
    // 3. 生成视频片段 (阿里百炼)
    // 4. 视频拼接 (FFmpeg)
    await this.taskQueue.add('story-generation', {
      taskId: task.id,
      dreamContent: params.dreamContent,
      stages: ['script', 'keyframes', 'videos', 'concat'],
    }, {
      // 长视频任务优先级较高，但允许更长的处理时间
      priority: 2,
      attempts: 3,
      timeout: 300000, // 5分钟超时
    });
    
    return task;
  }

  // 构建优化后的提示词
  private buildImagePrompt(params: ImageGenerationParams): string {
    // 根据梦境内容和风格构建提示词
    const stylePrompts = {
      '国潮': 'Chinese traditional art style, ink painting aesthetic',
      '赛博': 'cyberpunk style, neon lights, futuristic',
      '油画': 'oil painting style, classical art',
      '动漫': 'anime style, Japanese animation aesthetic',
      '科幻': 'sci-fi style, space, futuristic technology',
    };
    
    return `${params.dreamContent}, ${stylePrompts[params.style]}, dreamlike atmosphere, high quality, detailed`;
  }
}
```

#### 5.2.2 异步任务处理（简化版）

使用 Node.js 异步函数 + Redis 存储任务状态，替代重量级消息队列：

```typescript
// AI生成服务 - 简化版异步处理
class AIGenerationService {
  // 图片生成 - 直接异步处理
  async generateImage(params: ImageGenerationParams): Promise<void> {
    const taskId = await this.createTask(params);
    
    // 立即返回任务ID，前端轮询进度
    // 后台异步执行生成任务
    this.processImageGeneration(taskId, params).catch(error => {
      this.failTask(taskId, error);
    });
    
    return taskId;
  }
  
  // 图片生成处理逻辑
  private async processImageGeneration(taskId: string, params: ImageGenerationParams): Promise<void> {
    try {
      // 更新进度到Redis（前端轮询获取）
      await this.updateProgress(taskId, 10, '正在生成图片...');
      
      // 调用豆包API
      const result = await doubaoService.generateImage({
        prompt: params.prompt,
        style: params.style,
        size: '9:16',
      });
      
      await this.updateProgress(taskId, 60, '正在保存图片...');
      
      // 上传到Supabase Storage
      const imageUrl = await supabaseStorage.uploadFromUrl(result.imageUrl, {
        bucket: 'dream-images',
        path: `${params.userId}/${taskId}.png`,
      });
      
      await this.updateProgress(taskId, 90, '正在创建记录...');
      
      // 创建媒资记录
      await mediaService.createAsset({
        type: 'image',
        url: imageUrl,
        dreamId: params.dreamId,
        modelSource: 'doubao-seedream-4.5',
      });
      
      // 完成任务
      await this.completeTask(taskId, { imageUrl });
      await quotaService.consumeQuota(params.userId, 'image');
      
    } catch (error) {
      await this.failTask(taskId, error);
      throw error;
    }
  }
  
  // 梦境剧情生成 - 多阶段处理
  async generateStory(params: StoryGenerationParams): Promise<void> {
    const taskId = await this.createTask(params);
    
    // 后台异步执行多阶段任务
    this.processStoryGeneration(taskId, params).catch(error => {
      this.failTask(taskId, error);
    });
    
    return taskId;
  }
  
  private async processStoryGeneration(taskId: string, params: StoryGenerationParams): Promise<void> {
    const results: any = {};
    
    try {
      // 阶段1: 生成剧本
      await this.updateProgress(taskId, 10, '正在生成剧本...');
      results.script = await kimiService.generateScript(params.dreamContent, 4);
      await this.saveIntermediateResult(taskId, 'script', results.script);
      
      // 阶段2: 生成关键帧图
      await this.updateProgress(taskId, 30, '正在生成场景画面...');
      const keyframes = [];
      for (const scene of results.script.scenes) {
        const image = await doubaoService.generateImage({
          prompt: scene.description,
          style: '动漫',
        });
        keyframes.push(image);
      }
      results.keyframes = keyframes;
      await this.saveIntermediateResult(taskId, 'keyframes', keyframes);
      
      // 阶段3: 生成视频片段
      await this.updateProgress(taskId, 60, '正在生成视频片段...');
      const videoSegments = [];
      for (const keyframe of keyframes) {
        const video = await alibabaService.generateVideo({
          imageUrl: keyframe.url,
          duration: 5,
        });
        videoSegments.push(video);
      }
      results.videos = videoSegments;
      
      // 阶段4: 视频拼接
      await this.updateProgress(taskId, 90, '正在合成最终视频...');
      results.finalVideo = await ffmpegService.concatVideos(
        videoSegments.map(v => v.url),
        { outputDuration: 20 }
      );
      
      await this.completeTask(taskId, results);
      await quotaService.consumeQuota(params.userId, 'story');
      
    } catch (error) {
      await this.failTask(taskId, error);
      throw error;
    }
  }
  
  // 更新进度到Redis（前端轮询获取）
  private async updateProgress(taskId: string, progress: number, message: string): Promise<void> {
    await redis.setex(`task:${taskId}:progress`, 3600, JSON.stringify({
      progress,
      message,
      updatedAt: Date.now(),
    }));
  }
  
  // 获取任务进度（供前端轮询）
  async getTaskProgress(taskId: string): Promise<TaskProgress | null> {
    const data = await redis.get(`task:${taskId}:progress`);
    return data ? JSON.parse(data) : null;
  }
}
```

### 5.3 额度管理系统

```typescript
// 额度管理服务
class QuotaService {
  // 各用户类型的额度配置
  private quotaConfig = {
    guest: {        // 访客
      image: 1,
      video: 1,
      story: 0,
      interpretation: 1,
    },
    registered: {   // 注册用户
      image: 5,
      video: 2,
      story: 1,
      interpretation: Infinity,
    },
    subscribed: {   // 订阅用户
      image: Infinity,
      video: Infinity,
      story: Infinity,
      interpretation: Infinity,
    },
  };

  async checkQuota(userId: string, type: QuotaType): Promise<void> {
    const userProfile = await this.getUserProfile(userId);
    const limit = this.quotaConfig[userProfile.tier][type];
    
    if (limit === Infinity) return;
    
    const used = userProfile.usage[type];
    if (used >= limit) {
      throw new QuotaExceededError(type, limit, used);
    }
  }

  async consumeQuota(userId: string, type: QuotaType): Promise<void> {
    await supabase.rpc('increment_quota', {
      p_user_id: userId,
      p_quota_type: type,
    });
  }

  // 每日/每月重置额度（定时任务）
  async resetQuotas(period: 'daily' | 'monthly'): Promise<void> {
    // 仅重置非订阅用户的额度使用计数
    await supabase
      .from('user_profiles')
      .update({ [`usage_${period}`]: {} })
      .neq('tier', 'subscribed');
  }
}
```

```typescript
// 服务熔断器
class CircuitBreaker {
  private state: 'closed' | 'open' | 'half-open' = 'closed';
  private failureCount = 0;
  private lastFailureTime?: number;
  
  constructor(
    private threshold = 5,      // 失败阈值
    private timeout = 60000,    // 熔断持续时间(毫秒)
    private halfOpenMaxCalls = 3 // 半开状态最大测试请求数
  ) {}

  async execute<T>(fn: () => Promise<T>): Promise<T> {
    if (this.state === 'open') {
      if (Date.now() - (this.lastFailureTime || 0) > this.timeout) {
        this.state = 'half-open';
        this.failureCount = 0;
      } else {
        throw new Error('Circuit breaker is OPEN');
      }
    }

    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  private onSuccess(): void {
    this.failureCount = 0;
    if (this.state === 'half-open') {
      this.state = 'closed';
    }
  }

  private onFailure(): void {
    this.failureCount++;
    this.lastFailureTime = Date.now();
    
    if (this.failureCount >= this.threshold) {
      this.state = 'open';
    }
  }
}

// 为每个AI服务配置熔断器

  xunfei: new CircuitBreaker(),
  doubao: new CircuitBreaker(),
  alibaba: new CircuitBreaker(),
  kimi: new CircuitBreaker(),
};
```
